# Deployment Agent — Autonomyx LLM Gateway

## Role

An agentic workflow that deploys and configures the gateway autonomously.
Reads the skill, asks the right questions, generates all files, pushes to GitHub,
and triggers a Coolify deployment — end to end, no hand-holding.

Works as:
- A **Claude Code** session with bash + file tools
- A **Langflow** flow with HTTP + code nodes
- A **FastMCP tool** callable from any agent

---

## What the deployment agent does

```
1. GATHER    → collect inputs (providers, keys, deploy target, node config)
2. GENERATE  → produce all files from skill templates
3. VALIDATE  → check config.yaml syntax, env var completeness, port conflicts
4. PUSH      → commit to GitHub repo
5. DEPLOY    → trigger Coolify deployment via API
6. VERIFY    → health check all services, run test-tokens.sh
7. REPORT    → return deployment summary with endpoints and credentials
```

```python
"""
Autonomyx Gateway Deployment Agent

Autonomous end-to-end gateway deployment.
Requires: GitHub PAT, Coolify API key, target server UUID.

Usage:
    python agent.py --config deploy_config.json
    OR call via MCP tool: deploy_gateway(config)
"""

import os, json, subprocess, time, httpx, yaml
from pathlib import Path
from dataclasses import dataclass, field
from typing import Optional

# ── Config ────────────────────────────────────────────────────────────────

@dataclass

    github_org: str
    github_repo: str
    coolify_api_key: str
    coolify_server_uuid: str
    coolify_project_uuid: str

    groq_api_key: str = ""
    together_api_key: str = ""
    fireworks_api_key: str = ""
    openrouter_api_key: str = ""
    mistral_api_key: str = ""
    azure_api_key: str = ""
    azure_api_base: str = ""
    aws_access_key_id: str = ""
    aws_secret_access_key: str = ""
    aws_region: str = "ap-south-1"
    hf_token: str = ""

    # Deployment target
    deploy_target: str = "coolify"      # coolify | generic
    coolify_base_url: str = "http://vps.agnxxt.com:8000"
    domain_base: str = "openautonomyx.com"

    # Lago + Keycloak + Langfuse
    lago_db_password: str = "lago_pass"
    lago_secret_key_base: str = ""      # auto-generated if empty
    keycloak_admin_password: str = ""   # auto-generated if empty
    langfuse_db_password: str = "langfuse_pass"
    langfuse_nextauth_secret: str = ""  # auto-generated if empty
    mail_password: str = ""             # auto-generated if empty

    # Node config
    ollama_models: list = field(default_factory=lambda: [
        "qwen3:30b-a3b", "qwen2.5-coder:32b", "qwen2.5:14b",
        "llama3.1:8b", "mistral:7b-instruct", "llama3.2-vision:11b", "gemma3:9b"
    ])


# ── Step 1: Gather + validate ──────────────────────────────────────────────

def validate_config(cfg: DeployConfig) -> list[str]:
    errors = []
    if not cfg.github_pat:
        errors.append("github_pat required")
    if not cfg.litellm_master_key:
        errors.append("litellm_master_key required")
    if not cfg.coolify_api_key:
        errors.append("coolify_api_key required")

    active_providers = [k for k in [
        cfg.openai_api_key, cfg.anthropic_api_key, cfg.gemini_api_key,
        cfg.groq_api_key
    ] if k]
    if not active_providers:
        errors.append("At least one cloud provider API key required (or use local-only mode)")

    return errors


def auto_generate_secrets(cfg: DeployConfig) -> DeployConfig:
    """Fill in any missing secrets with secure random values."""
    import secrets
    if not cfg.lago_secret_key_base:
        cfg.lago_secret_key_base = secrets.token_hex(32)
    if not cfg.keycloak_admin_password:
        cfg.keycloak_admin_password = secrets.token_urlsafe(16)
    if not cfg.langfuse_nextauth_secret:
        cfg.langfuse_nextauth_secret = secrets.token_urlsafe(32)
    if not cfg.mail_password:
        cfg.mail_password = secrets.token_urlsafe(16)
    return cfg


# ── Step 2: Generate files ─────────────────────────────────────────────────

def generate_config_yaml(cfg: DeployConfig) -> str:
    """

    """
    models = []

    # Always-on local models
    for model in cfg.ollama_models:

            "model_name": f"ollama/{model}",
            "litellm_params": {
                "model": f"ollama/{model}",
                "api_base": "http://ollama:11434",
            }
        })

    # Cloud providers — only if key present
    if cfg.openai_api_key:
        models += [
            {"model_name": "gpt-4o",
             "litellm_params": {"model": "gpt-4o", "api_key": "os.environ/OPENAI_API_KEY"}},
            {"model_name": "gpt-4o-mini",
             "litellm_params": {"model": "gpt-4o-mini", "api_key": "os.environ/OPENAI_API_KEY"}},
        ]
    if cfg.anthropic_api_key:
        models += [
            {"model_name": "claude-3-5-sonnet",
             "litellm_params": {"model": "anthropic/claude-sonnet-4-5",
                                "api_key": "os.environ/ANTHROPIC_API_KEY"}},
            {"model_name": "claude-3-haiku",
             "litellm_params": {"model": "anthropic/claude-haiku-4-5-20251001",
                                "api_key": "os.environ/ANTHROPIC_API_KEY"}},
        ]
    if cfg.groq_api_key:
        models.append(
            {"model_name": "groq/llama3-70b",
             "litellm_params": {"model": "groq/llama3-70b-8192",
                                "api_key": "os.environ/GROQ_API_KEY"}},
        )

    config = {
        "model_list": models,
        "router_settings": {
            "routing_strategy": "usage-based-routing",
            "num_retries": 3,
            "request_timeout": 60,
            "allowed_fails": 2,

                {"ollama/qwen3:30b-a3b": ["groq/llama3-70b", "claude-3-5-sonnet"]},
                {"ollama/qwen2.5-coder:32b": ["groq/llama3-70b", "gpt-4o"]},
            ]
        },
        "litellm_settings": {
            "drop_params": True,
            "max_budget": 10.0,
            "budget_duration": "30d",
            "store_model_in_db": True,
            "success_callback": ["lago_callback.LagoCallback"],
        },
        "general_settings": {
            "master_key": "os.environ/LITELLM_MASTER_KEY",
            "database_url": "os.environ/DATABASE_URL",
            "store_model_in_db": True,
            "additional_routers": ["recommender.router", "feedback.router"],
        }
    }
    return yaml.dump(config, default_flow_style=False, sort_keys=False)


# ── Step 3: Validate generated files ──────────────────────────────────────

def validate_config_yaml(config_str: str) -> list[str]:
    """Parse and validate the generated config.yaml."""
    errors = []
    try:
        config = yaml.safe_load(config_str)
        if not config.get("model_list"):
            errors.append("config.yaml: model_list is empty")
        if not config.get("general_settings", {}).get("master_key"):
            errors.append("config.yaml: master_key missing")
        for model in config.get("model_list", []):
            if not model.get("model_name"):
                errors.append(f"config.yaml: model entry missing model_name: {model}")
    except yaml.YAMLError as e:
        errors.append(f"config.yaml: YAML parse error: {e}")
    return errors


# ── Step 4: Push to GitHub ─────────────────────────────────────────────────

def push_to_github(
    cfg: DeployConfig,
    files: dict[str, str],
    commit_message: str,
    branch: str = "main",
) -> str:
    """
    Push generated files to GitHub repo.
    files: {filename: content}
    Returns commit SHA.
    """
    base_url = f"https://api.github.com/repos/{cfg.github_org}/{cfg.github_repo}"
    headers = {
        "Authorization": f"Bearer {cfg.github_pat}",
        "Accept": "application/vnd.github+json",
    }

    commit_sha = None
    for filename, content in files.items():
        import base64
        encoded = base64.b64encode(content.encode()).decode()

        # Check if file exists (to get SHA for update)
        r = httpx.get(f"{base_url}/contents/{filename}", headers=headers)
        payload = {
            "message": commit_message,
            "content": encoded,
            "branch": branch,
        }
        if r.status_code == 200:
            payload["sha"] = r.json()["sha"]  # required for update

        r = httpx.put(f"{base_url}/contents/{filename}", headers=headers, json=payload)
        r.raise_for_status()
        commit_sha = r.json()["commit"]["sha"]

    return commit_sha


# ── Step 5: Deploy via Coolify ─────────────────────────────────────────────

def trigger_coolify_deploy(cfg: DeployConfig, service_uuid: str) -> dict:
    """Trigger a Coolify service redeployment."""
    r = httpx.post(
        f"{cfg.coolify_base_url}/api/v1/services/{service_uuid}/restart",
        headers={
            "Authorization": f"Bearer {cfg.coolify_api_key}",
            "Content-Type": "application/json",
        },
        timeout=30,
    )
    r.raise_for_status()
    return r.json()


# ── Step 6: Health check ───────────────────────────────────────────────────

def health_check(gateway_url: str, master_key: str, retries: int = 10) -> bool:
    """Poll gateway health endpoint until ready."""
    for i in range(retries):
        try:
            r = httpx.get(
                f"{gateway_url}/health",
                headers={"Authorization": f"Bearer {master_key}"},
                timeout=10,
            )
            if r.status_code == 200:
                return True
        except Exception:
            pass
        print(f"Health check {i+1}/{retries} — waiting...")
        time.sleep(15)
    return False


# ── Step 7: Report ────────────────────────────────────────────────────────

def build_report(cfg: DeployConfig, commit_sha: str, healthy: bool) -> dict:
    return {
        "status": "deployed" if healthy else "deployed_unhealthy",
        "commit_sha": commit_sha,
        "endpoints": {
            "gateway":   f"https://llm.{cfg.domain_base}",
            "grafana":   f"https://metrics.{cfg.domain_base}",
            "langfuse":  f"https://traces.{cfg.domain_base}",
            "lago_ui":   f"https://billing-ui.{cfg.domain_base}",
            "keycloak":  f"https://auth.{cfg.domain_base}",
            "mcp":       f"https://mcp.{cfg.domain_base}/sse",
        },
        "credentials": {
            "litellm_master_key": cfg.litellm_master_key,
            "keycloak_admin":     cfg.keycloak_admin_password,
            "grafana_admin":      "autonomyx",
        },
        "next_steps": [
            "1. Import Grafana dashboard ID 17587",
            "2. Create Lago billable metrics (llm_input_tokens, llm_output_tokens, llm_requests)",
            "3. Pull always-on Ollama models: docker exec autonomyx-ollama sh /ollama-pull.sh",
            "4. Create first Keycloak tenant group to provision Lago customer + LiteLLM key",
            "5. Test: curl https://llm.{domain}/health",
        ],
    }


# ── Main orchestration ─────────────────────────────────────────────────────

def deploy(config_path: str) -> dict:
    """Full deployment pipeline."""

    # Load config
    with open(config_path) as f:
        raw = json.load(f)
    cfg = DeployConfig(**raw)

    print("Step 1: Validating config...")
    errors = validate_config(cfg)
    if errors:
        return {"status": "error", "errors": errors}
    cfg = auto_generate_secrets(cfg)

    print("Step 2: Generating files...")
    env_content     = generate_env_file(cfg)
    config_content  = generate_config_yaml(cfg)

    print("Step 3: Validating generated config.yaml...")
    yaml_errors = validate_config_yaml(config_content)
    if yaml_errors:
        return {"status": "error", "errors": yaml_errors}

    print("Step 4: Pushing to GitHub...")
    commit_sha = push_to_github(
        cfg,
        files={
            ".env": env_content,
            "config.yaml": config_content,
        },
        commit_message="chore: gateway config update via deployment agent",
    )
    print(f"  Committed: {commit_sha}")

    print("Step 5: Triggering Coolify deploy...")
    # trigger_coolify_deploy(cfg, service_uuid="YOUR_SERVICE_UUID")

    print("Step 6: Health check...")
    gateway_url = f"https://llm.{cfg.domain_base}"
    healthy = health_check(gateway_url, cfg.litellm_master_key)

    print("Step 7: Building report...")
    report = build_report(cfg, commit_sha, healthy)

    print("\n=== DEPLOYMENT COMPLETE ===")
    print(json.dumps(report, indent=2))
    return report


if __name__ == "__main__":
    import sys
    config_path = sys.argv[1] if len(sys.argv) > 1 else "deploy_config.json"
    deploy(config_path)
```

## As an MCP tool (callable from Claude / other agents)

```python
# Register deploy_gateway as an MCP tool in autonomyx-mcp

@mcp.tool()
async def deploy_gateway(config: dict) -> dict:
    """
    Deploy or update the Autonomyx LLM Gateway autonomously.

    Generates all config files, pushes to GitHub, triggers Coolify deployment,
    health checks all services, returns endpoints and credentials.

    Args:
        config: DeployConfig dict (see deploy_config.json template)

    Returns:
        status, endpoints, credentials, next_steps
    """
    import tempfile, json
    with tempfile.NamedTemporaryFile(mode='w', suffix='.json', delete=False) as f:
        json.dump(config, f)
        config_path = f.name
    from deploy_agent.agent import deploy
    return deploy(config_path)
```
